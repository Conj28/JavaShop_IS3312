DROP TABLE "USERDATA";

CREATE TABLE "USERDATA"
(
"USER_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"EMAIL" VARCHAR(50), 
"PASSWORD" VARCHAR(50), 
"FNAME" VARCHAR(50),   
"LNAME" VARCHAR(50),
"USERTYPE" VARCHAR(10)
);

DROP TABLE "PROD";

CREATE TABLE "PROD"
(
"PRODUCT_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
"NAME" VARCHAR(50),
"IMG" VARCHAR(10000),
"PRICE" VARCHAR(10),
"DESCRIPTION" VARCHAR(500),
"DIMENSIONS" VARCHAR(50),
"MATERIAL" VARCHAR(50),
"TYPE" VARCHAR(50)
);

DROP TABLE "CART";

CREATE TABLE "CART"
(
"CART_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),
"PRODUCT_ID" INT,
"NAME" VARCHAR(50),
"IMG" VARCHAR(10000),
"PRICE" INT,
"QUANTITY" INT
);
